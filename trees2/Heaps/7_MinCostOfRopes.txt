long long minCost(long long arr[], long long n) {
    if(n==1) return 0;
    priority_queue<long long,vector<long long> , greater<long long> > pq;
    for(long long i=0;i<n;i++){
        pq.push(arr[i]);
    }
    long long ans = 0;
    while(pq.size()>1){
        long long l1 = pq.top();
        pq.pop();
        long long l2 = pq.top();
        pq.pop();
        long long l3 = l1 + l2;
        ans+=l3;
        pq.push(l3);
    }
    return ans;
}
